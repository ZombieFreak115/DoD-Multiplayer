country_event = {
	id = 2020
	title = "EVTNAME2020"
	desc = "EVTDESC2020_NEW"
	picture = "emigration"
	
	trigger = {
		any_owned_province = {
			OR = {
				continent = north_america
				continent = south_america
				continent = oceania
			}
			any_pop = {
				NOT = {
					is_culture_group = oceanic
					is_culture_group = native_american
					is_culture_group = american_cultures
					is_culture_group = neo_european_cultures
					is_culture_group = latin_american_cultures
					is_culture_group = african_diaspora_cultures
				}
			}
			is_colonial = no
		}
		civilized = yes
		NOT = {
			religious_policy = moralism
			citizenship_policy = residency
		}
	}
	
	mean_time_to_happen = {
		months = 250
		modifier = {
			factor = 0.9
			citizenship_policy = full_citizenship
		}
		modifier = {
			factor = 0.9
			average_consciousness = 3
		}
		modifier = {
			factor = 0.9
			average_consciousness = 4
		}
		modifier = {

			plurality = 20
		}

			factor = 0.9
			plurality = 25
		}
		modifier = {
			factor = 0.9
			plurality = 30
		}
	}
	
	option = {
		name = "EVTOPTA2020"
		random_owned = {
			limit = {
				OR = {
					continent = north_america
					continent = south_america
					continent = oceania
				}
				any_pop = {
					NOT = {
						is_culture_group = oceanic
						is_culture_group = native_american
						is_culture_group = american_cultures
						is_culture_group = neo_european_cultures
						is_culture_group = latin_american_cultures

				}
				is_colonial = no

			any_pop = {
				militancy = -1
			}
		}

		name = "EVTOPTB2020"
